# sine & cosine curve
x=np.arange(0, 5*np.pi,0.1)
y_sin =np.sin(x)
y_cos =np.cos(x)

#setup subplot grid , height is 2 and width is 1
plt.subplot(2,1,1)

#make first plot
plt.plot(x, y_sin, 'r*--')
plt.title('sine')

#second subplot as active & make it
plt.subplot(2,1,2)
plt.plot(x, y_cos, 'g--')
plt.title("cosine")

plt.show()
